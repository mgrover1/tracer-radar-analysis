import pyart
import act
import glob
import matplotlib.pyplot as plt
from pathlib import Path

Grab Data#

june2_csapr_files = act.discovery.download_data(
    "######", "#####", "houcsapr2cfrS2.a1", "2022-06-02", "2022-06-02"
)

Read in and Plot the Data#

Before following running the next cells, make sure you have created the following directories:

  • quicklooks/ppi

  • quicklooks/rhi

  • quicklooks/vpt

radar_files = sorted(glob.glob("houcsapr2cfrS2.a1/houcsapr2cfrS2.a1.20220602*"))

for file in radar_files:
    print(file)
    try:
        radar = pyart.io.read(file)
    except:
        continue
    print(radar.scan_type)
    if radar.scan_type == 'other':
        radar.scan_type = 'vpt'
    display = pyart.graph.RadarDisplay(radar)
    display.plot("reflectivity", 0)
    plt.savefig(f"quicklooks/{radar.scan_type}/{Path(file).stem}.png", dpi=200)
    plt.show()
    plt.close()
